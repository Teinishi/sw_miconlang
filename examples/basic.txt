microcontroller BasicExample {
    name = "Basic Example"
    description = "description of the microcontroller"

    interface {
        inputs {
            a: float {
                name = "Input A"
            }
            b: float
        }
        outputs {
            c: float {
                name = "Output C"
                position = (0, 0)
            }
        }
    }

    logic {
        outputs.c = inputs.a + 2 * inputs.b
    }
}
